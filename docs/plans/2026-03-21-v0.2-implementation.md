### Task 6: BCD SBC N/Z Flag Fix — Implementation

**Files:**
- Modify: `components/apple2/cpu6502.c:259-285` (cpu_sbc function)

**Step 1: Set N/Z from binary result before BCD adjustment**

Replace the `cpu_sbc` function body. The key change: `cpu_set_nz` is called on `(uint8_t)diff` (binary result) before the decimal branch, and we set `cpu->a` to the binary result first, then overwrite with BCD result only in decimal mode:

```c
static void cpu_sbc(cpu6502_t *cpu, uint8_t value)
{
    const uint8_t carry = cpu_flag(cpu, CPU6502_FLAG_CARRY) ? 1U : 0U;
    const uint8_t a = cpu->a;
    const uint16_t diff = (uint16_t)a - value - (uint16_t)(1U - carry);

    cpu_set_flag(cpu, CPU6502_FLAG_OVERFLOW,
                 ((a ^ value) & (uint8_t)(a ^ diff) & 0x80U) != 0U);
    cpu_set_flag(cpu, CPU6502_FLAG_CARRY, diff < 0x100U);
    cpu_set_nz(cpu, (uint8_t)diff);

    if (cpu_flag(cpu, CPU6502_FLAG_DECIMAL)) {
        int16_t lo = (int16_t)(a & 0x0FU) - (int16_t)(value & 0x0FU) - (int16_t)(1U - carry);
        int16_t hi = (int16_t)(a >> 4) - (int16_t)(value >> 4);
        if (lo < 0) {
            lo -= 0x06;
            hi--;
        }
        if (hi < 0) {
            hi -= 0x06;
        }
        cpu->a = (uint8_t)(((uint8_t)(hi << 4) & 0xF0U) | ((uint8_t)lo & 0x0FU));
    } else {
        cpu->a = (uint8_t)diff;
    }
}
```

### Task 8: Disk Nibble Multi-Advance — Implementation

**Files:**
- Modify: `components/apple2/apple2_disk2.c:774-809` (apple2_disk2_tick)

**Step 1: Change single-advance to while loop with cap**

Replace the body of `apple2_disk2_tick` from line 791 onwards:

```c
void apple2_disk2_tick(apple2_disk2_t *disk2, uint32_t cpu_hz, uint32_t cycles)
{
    uint8_t drive;
    uint32_t *stream_accum;
    uint32_t accum;
    uint32_t *nibble_pos;
    uint16_t track_length;
    unsigned advances = 0;

    if (disk2 == NULL || cycles == 0U || cpu_hz == 0U || !disk2->motor_on) {
        return;
    }
    drive = disk2->active_drive;
    if (!disk2->loaded[drive]) {
        return;
    }

    stream_accum = &disk2->stream_accum[drive];
    accum = *stream_accum + (APPLE2_DISK2_BYTES_PER_SECOND * cycles);
    if (accum < cpu_hz) {
        *stream_accum = accum;
        return;
    }
    *stream_accum = accum;
    if (!disk2_prepare_track(disk2)) {
        return;
    }
    track_length = disk2->track_cache_length;
    nibble_pos = &disk2->nibble_pos[drive];

    while (*stream_accum >= cpu_hz && advances < 8U) {
        *stream_accum -= cpu_hz;
        (*nibble_pos)++;
        if (*nibble_pos >= track_length) {
            *nibble_pos = 0U;
        }
        advances++;
    }
    disk2_latch_prepared_byte(disk2);
}
```

### Task 11: Audio Implementation — Stub

**Files:**
- Create: `components/cardputer/cardputer_audio.c`

**Step 1: Write the implementation with I2S output**

This is an ESP-IDF-only file (won't compile in the host test harness). The implementation:

- Allocates a `cardputer_audio_t` struct with a circular toggle ring buffer (64 entries of `uint64_t` cycle stamps), output level state, and I2S handle.
- `cardputer_audio_init()`: Configures I2S standard mode (16-bit mono, sample rate from Kconfig). For the Original variant, also initializes the ES8311 codec over I2C. For the ADV (NS4168), just I2S config.
- `cardputer_audio_toggle()`: Appends the cycle count to the ring buffer.
- `cardputer_audio_flush()`: Walks from the last flushed cycle to `cpu_cycle`, computing sample boundaries at `cpu_hz / sample_rate` cycles per sample. At each toggle in the ring, flips the output level. Fills an I2S write buffer with the constant level between toggles. Calls `i2s_channel_write()` (non-blocking, best-effort).
- `cardputer_audio_deinit()`: Disables and frees the I2S channel.

The full implementation depends on the ESP-IDF I2S API and the specific I2C register map for the ES8311. These are hardware-specific and should be written with the ESP-IDF documentation at hand. The structure above provides the complete design; the implementer should use `i2s_std_config_t` with `I2S_DATA_BIT_WIDTH_16BIT` and `I2S_SLOT_MODE_MONO`.

**Step 2: Implement apple2_disk2_flush**

```c
bool apple2_disk2_flush(apple2_disk2_t *disk2)
{
    if (!disk2->track_cache_dirty || !disk2->track_cache_valid) {
        return true;
    }

    const uint8_t drive = disk2->track_cache_drive;
    if (drive >= 2U || disk2->write_sector[drive] == NULL) {
        return false;
    }

    const apple2_disk2_image_order_t order = disk2->image_order[drive];
    const uint8_t *track_order =
        (order == APPLE2_DISK2_IMAGE_ORDER_DOS33_LOGICAL)
            ? s_dos33_track_order : s_prodos_track_order;

    disk2_ensure_gcr62_inverse();

    const uint16_t len = disk2->track_cache_length;
    const uint8_t *cache = disk2->track_cache;
    bool ok = true;

    for (uint16_t i = 0; i + 3U <= len; ++i) {
        /* Find address field: D5 AA 96 */
        if (cache[i] != 0xD5U || cache[i + 1U] != 0xAAU || cache[i + 2U] != 0x96U) {
            continue;
        }

        const uint16_t hdr = (uint16_t)(i + 3U);
        if (hdr + 10U > len) break;

        const uint8_t volume = test_decode44_inline(cache[hdr], cache[hdr + 1U]);
        const uint8_t track = test_decode44_inline(cache[hdr + 2U], cache[hdr + 3U]);
        const uint8_t phys_sector = test_decode44_inline(cache[hdr + 4U], cache[hdr + 5U]);
        (void)volume;

        /* Find data field: D5 AA AD */
        uint16_t j = (uint16_t)(hdr + 8U);
        bool found_data = false;
        while (j + 3U <= len && j < (uint16_t)(hdr + 60U)) {
            if (cache[j] == 0xD5U && cache[j + 1U] == 0xAAU && cache[j + 2U] == 0xADU) {
                found_data = true;
                break;
            }
            j++;
        }
        if (!found_data) continue;

        const uint16_t data_start = (uint16_t)(j + 3U);
        if (data_start + 343U > len) continue;

        /* Reverse 6-and-2 GCR decode. */
        uint8_t decoded[342];
        uint8_t prev = 0;
        for (uint16_t k = 0; k < 342U; ++k) {
            const uint8_t raw = s_disk2_gcr62_inverse[cache[data_start + k]];
            decoded[k] = (uint8_t)(prev ^ raw);
            prev = decoded[k];
        }

        /* Reconstruct 256-byte sector from aux[86] + upper[256]. */
        uint8_t sector_data[256];
        for (uint16_t si = 0; si < 256U; ++si) {
            sector_data[si] = (uint8_t)(decoded[86U + si] << 2);
        }
        for (uint16_t si = 0; si < 86U; ++si) {
            const uint16_t aux_index = 85U - si;
            const uint8_t aux = decoded[aux_index];
            const uint8_t low2_0 = (uint8_t)(((aux & 0x02U) >> 1) | ((aux & 0x01U) << 1));
            sector_data[si] |= low2_0;
            if (si + 86U < 256U) {
                const uint8_t low2_1 = (uint8_t)(((aux & 0x08U) >> 3) | ((aux & 0x04U) >> 1));
                sector_data[si + 86U] |= low2_1;
            }
            if (si + 172U < 256U) {
                const uint8_t low2_2 = (uint8_t)(((aux & 0x20U) >> 5) | ((aux & 0x10U) >> 3));
                sector_data[si + 172U] |= low2_2;
            }
        }

        const uint8_t file_sector = disk2_find_file_sector_for_physical(track_order, phys_sector);
        if (!disk2->write_sector[drive](disk2->write_sector_context[drive],
                                         drive, track, file_sector, sector_data)) {
            ok = false;
        }

        i = (uint16_t)(data_start + 342U);
    }

    if (ok) {
        disk2->track_cache_dirty = false;
    }
    return ok;
}
```
